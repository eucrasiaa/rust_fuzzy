100    /// Add a constraint to the solver.
101    pub fn add_constraint(&mut self, constraint: Constraint) -> Result<(), AddConstraintError> {
102        if self.constraints.contains_key(&constraint) {
103            // TODO detrmine if we could just ignore duplicate constraints
104            return Err(AddConstraintError::DuplicateConstraint);
105        }
106
107        // Creating a row causes symbols to reserved for the variables in the constraint. If this
108        // method exits with an exception, then its possible those variables will linger in the var
109        // map. Since its likely that those variables will be used in other constraints and since
110        // exceptional conditions are uncommon, i'm not too worried about aggressive cleanup of the
111        // var map.
112        let (mut row, tag) = self.create_row(&constraint);
113        let mut subject = Solver::choose_subject(&row, &tag);
114
115        // If choose_subject could find a valid entering symbol, one last option is available if the
116        // entire row is composed of dummy variables. If the constant of the row is zero, then this
117        // represents redundant constraints and the new dummy marker can enter the basis. If the
118        // constant is non-zero, then it represents an unsatisfiable constraint.
119        if subject.kind() == SymbolKind::Invalid && Solver::all_dummies(&row) {
120            if !near_zero(row.constant) {
121                return Err(AddConstraintError::UnsatisfiableConstraint);
122            } else {
123                subject = tag.marker;
124            }
125        }
126
127        // If an entering symbol still isn't found, then the row must be added using an artificial
128        // variable. If that fails, then the row represents an unsatisfiable constraint.
129        if subject.kind() == SymbolKind::Invalid {
130            let satisfiable = self.add_with_artificial_variable(&row)?;
131            if !satisfiable {
132                return Err(AddConstraintError::UnsatisfiableConstraint);
133            }
134        } else {
135            row.solve_for_symbol(subject);
136            self.substitute(subject, &row);
137            if subject.kind() == SymbolKind::External && row.constant != 0.0 {
138                let v = self.var_for_symbol[&subject];
139                self.var_changed(v);
140            }
141            self.rows.insert(subject, row);
142        }
143
144        self.constraints.insert(constraint, tag);
145
146        // Optimizing after each constraint is added performs less aggregate work due to a smaller
147        // average system size. It also ensures the solver remains in a consistent state.
148        self.optimize(OptimizeTarget::Objective)?;
149        Ok(())
150    }

152    /// Remove a constraint from the solver.
153    pub fn remove_constraint(
154        &mut self,
155        constraint: &Constraint,
156    ) -> Result<(), RemoveConstraintError> {
157        let tag = self
158            .constraints
159            .remove(constraint)
160            .ok_or(RemoveConstraintError::UnknownConstraint)?;
161
162        // Remove the error effects from the objective function
163        // *before* pivoting, or substitutions into the objective
164        // will lead to incorrect solver results.
165        self.remove_constraint_effects(constraint, &tag);
166
167        // If the marker is basic, simply drop the row. Otherwise,
168        // pivot the marker into the basis and then drop the row.
169        if self.rows.remove(&tag.marker).is_none() {
170            let (leaving, mut row) = self.get_marker_leaving_row(tag.marker).ok_or(
171                RemoveConstraintError::InternalSolverError(
172                    InternalSolverError::FailedToFindLeavingRow,
173                ),
174            )?;
175            row.solve_for_symbols(leaving, tag.marker);
176            self.substitute(tag.marker, &row);
177        }
178
179        // Optimizing after each constraint is removed ensures that the
180        // solver remains consistent. It makes the solver api easier to
181        // use at a small tradeoff for speed.
182        self.optimize(OptimizeTarget::Objective)?;
183
184        // Check for and decrease the reference count for variables referenced by the constraint
185        // If the reference count is zero remove the variable from the variable map
186        for term in &constraint.expr().terms {
187            if !near_zero(term.coefficient) {
188                let mut should_remove = false;
189                if let Some(&mut (_, _, ref mut count)) = self.var_data.get_mut(&term.variable) {
190                    *count -= 1;
191                    should_remove = *count == 0;
192                }
193                if should_remove {
194                    self.var_for_symbol.remove(&self.var_data[&term.variable].1);
195                    self.var_data.remove(&term.variable);
196                }
197            }
198        }
199        Ok(())
200    }

207    /// Add an edit variable to the solver.
208    ///
209    /// This method should be called before the `suggest_value` method is
210    /// used to supply a suggested value for the given edit variable.
211    pub fn add_edit_variable(
212        &mut self,
213        v: Variable,
214        strength: Strength,
215    ) -> Result<(), AddEditVariableError> {
216        if self.edits.contains_key(&v) {
217            return Err(AddEditVariableError::DuplicateEditVariable);
218        }
219        if strength == Strength::REQUIRED {
220            return Err(AddEditVariableError::BadRequiredStrength);
221        }
222        let cn = Constraint::new(
223            Expression::from_term(Term::new(v, 1.0)),
224            RelationalOperator::Equal,
225            strength,
226        );
227        self.add_constraint(cn.clone()).unwrap();
228        self.edits.insert(
229            v,
230            EditInfo {
231                tag: self.constraints[&cn],
232                constraint: cn,
233                constant: 0.0,
234            },
235        );
236        Ok(())
237    }

239    /// Remove an edit variable from the solver.
240    pub fn remove_edit_variable(&mut self, v: Variable) -> Result<(), RemoveEditVariableError> {
241        if let Some(constraint) = self.edits.remove(&v).map(|e| e.constraint) {
242            self.remove_constraint(&constraint).map_err(|e| match e {
243                RemoveConstraintError::UnknownConstraint => {
244                    RemoveEditVariableError::InternalSolverError(
245                        InternalSolverError::EditConstraintNotInSystem,
246                    )
247                }
248                RemoveConstraintError::InternalSolverError(s) => {
249                    RemoveEditVariableError::InternalSolverError(s)
250                }
251            })?;
252            Ok(())
253        } else {
254            Err(RemoveEditVariableError::UnknownEditVariable)
255        }
256    }

263    /// Suggest a value for the given edit variable.
264    ///
265    /// This method should be used after an edit variable has been added to
266    /// the solver in order to suggest the value for that variable.
267    pub fn suggest_value(
268        &mut self,
269        variable: Variable,
270        value: f64,
271    ) -> Result<(), SuggestValueError> {
272        let (info_tag_marker, info_tag_other, delta) = {
273            let info = self
274                .edits
275                .get_mut(&variable)
276                .ok_or(SuggestValueError::UnknownEditVariable)?;
277            let delta = value - info.constant;
278            info.constant = value;
279            (info.tag.marker, info.tag.other, delta)
280        };
281        // tag.marker and tag.other are never external symbols
282
283        // The nice version of the following code runs into non-lexical borrow issues.
284        // Ideally the `if row...` code would be in the body of the if. Pretend that it is.
285        {
286            let infeasible_rows = &mut self.infeasible_rows;
287            if self
288                .rows
289                .get_mut(&info_tag_marker)
290                .map(|row| {
291                    if row.add(-delta) < 0.0 {
292                        infeasible_rows.push(info_tag_marker);
293                    }
294                })
295                .is_some()
296                || self
297                    .rows
298                    .get_mut(&info_tag_other)
299                    .map(|row| {
300                        if row.add(delta) < 0.0 {
301                            infeasible_rows.push(info_tag_other);
302                        }
303                    })
304                    .is_some()
305            {
306            } else {
307                for (symbol, row) in &mut self.rows {
308                    let coeff = row.coefficient_for(info_tag_marker);
309                    let diff = delta * coeff;
310                    if diff != 0.0 && symbol.kind() == SymbolKind::External {
311                        let v = self.var_for_symbol[symbol];
312                        // inline var_changed - borrow checker workaround
313                        if self.should_clear_changes {
314                            self.changed.clear();
315                            self.should_clear_changes = false;
316                        }
317                        self.changed.insert(v);
318                    }
319                    if coeff != 0.0 && row.add(diff) < 0.0 && symbol.kind() != SymbolKind::External
320                    {
321                        infeasible_rows.push(*symbol);
322                    }
323                }
324            }
325        }
326        self.dual_optimize()?;
327        Ok(())
328    }

338    /// Fetches all changes to the values of variables since the last call to this function.
339    ///
340    /// The list of changes returned is not in a specific order. Each change comprises the variable
341    /// changed and the new value of that variable.
342    pub fn fetch_changes(&mut self) -> &[(Variable, f64)] {
343        if self.should_clear_changes {
344            self.changed.clear();
345            self.should_clear_changes = false;
346        } else {
347            self.should_clear_changes = true;
348        }
349        self.public_changes.clear();
350        for &v in &self.changed {
351            if let Some(var_data) = self.var_data.get_mut(&v) {
352                let new_value = self
353                    .rows
354                    .get(&var_data.1)
355                    .map(|r| r.constant)
356                    .unwrap_or(0.0);
357                let old_value = var_data.0;
358                if old_value != new_value {
359                    self.public_changes.push((v, new_value));
360                    var_data.0 = new_value;
361                }
362            }
363        }
364        &self.public_changes
365    }

388    /// Get the symbol for the given variable.
389    ///
390    /// If a symbol does not exist for the variable, one will be created.
391    fn get_var_symbol(&mut self, v: Variable) -> Symbol {
392        let id_tick = &mut self.id_tick;
393        let var_for_symbol = &mut self.var_for_symbol;
394        let value = self.var_data.entry(v).or_insert_with(|| {
395            let s = Symbol::new(*id_tick, SymbolKind::External);
396            var_for_symbol.insert(s, v);
397            *id_tick += 1;
398            (f64::NAN, s, 0)
399        });
400        value.2 += 1;
401        value.1
402    }

404    /// Create a new Row object for the given constraint.
405    ///
406    /// The terms in the constraint will be converted to cells in the row. Any term in the
407    /// constraint with a coefficient of zero is ignored. This method uses the `get_var_symbol`
408    /// method to get the symbol for the variables added to the row. If the symbol for a given cell
409    /// variable is basic, the cell variable will be substituted with the basic row.
410    ///
411    /// The necessary slack and error variables will be added to the row. If the constant for the
412    /// row is negative, the sign for the row will be inverted so the constant becomes positive.
413    ///
414    /// The tag will be updated with the marker and error symbols to use for tracking the movement
415    /// of the constraint in the tableau.
416    fn create_row(&mut self, constraint: &Constraint) -> (Box<Row>, Tag) {
417        let expr = constraint.expr();
418        let mut row = Row::new(expr.constant);
419
420        // Substitute the current basic variables into the row.
421        for term in &expr.terms {
422            if !near_zero(term.coefficient) {
423                let symbol = self.get_var_symbol(term.variable);
424                if let Some(other_row) = self.rows.get(&symbol) {
425                    row.insert_row(other_row, term.coefficient);
426                } else {
427                    row.insert_symbol(symbol, term.coefficient);
428                }
429            }
430        }
431
432        let objective = &mut self.objective;
433
434        // Add the necessary slack, error, and dummy variables.
435        let tag = match constraint.op() {
436            RelationalOperator::GreaterOrEqual | RelationalOperator::LessOrEqual => {
437                let coeff = if constraint.op() == RelationalOperator::LessOrEqual {
438                    1.0
439                } else {
440                    -1.0
441                };
442                let slack = Symbol::new(self.id_tick, SymbolKind::Slack);
443                self.id_tick += 1;
444                row.insert_symbol(slack, coeff);
445                if constraint.strength() < Strength::REQUIRED {
446                    let error = Symbol::new(self.id_tick, SymbolKind::Error);
447                    self.id_tick += 1;
448                    row.insert_symbol(error, -coeff);
449                    objective.insert_symbol(error, constraint.strength().value());
450                    Tag {
451                        marker: slack,
452                        other: error,
453                    }
454                } else {
455                    Tag {
456                        marker: slack,
457                        other: Symbol::invalid(),
458                    }
459                }
460            }
461            RelationalOperator::Equal => {
462                if constraint.strength() < Strength::REQUIRED {
463                    let errplus = Symbol::new(self.id_tick, SymbolKind::Error);
464                    self.id_tick += 1;
465                    let errminus = Symbol::new(self.id_tick, SymbolKind::Error);
466                    self.id_tick += 1;
467                    row.insert_symbol(errplus, -1.0); // v = eplus - eminus
468                    row.insert_symbol(errminus, 1.0); // v - eplus + eminus = 0
469                    objective.insert_symbol(errplus, constraint.strength().value());
470                    objective.insert_symbol(errminus, constraint.strength().value());
471                    Tag {
472                        marker: errplus,
473                        other: errminus,
474                    }
475                } else {
476                    let dummy = Symbol::new(self.id_tick, SymbolKind::Dummy);
477                    self.id_tick += 1;
478                    row.insert_symbol(dummy, 1.0);
479                    Tag {
480                        marker: dummy,
481                        other: Symbol::invalid(),
482                    }
483                }
484            }
485        };
486
487        // Ensure the row has a positive constant.
488        if row.constant < 0.0 {
489            row.reverse_sign();
490        }
491        (Box::new(row), tag)
492    }

494    /// Choose the subject for solving for the row.
495    ///
496    /// This method will choose the best subject for using as the solve
497    /// target for the row. An invalid symbol will be returned if there
498    /// is no valid target.
499    ///
500    /// The symbols are chosen according to the following precedence:
501    ///
502    /// 1) The first symbol representing an external variable.
503    /// 2) A negative slack or error tag variable.
504    ///
505    /// If a subject cannot be found, an invalid symbol will be returned.
506    fn choose_subject(row: &Row, tag: &Tag) -> Symbol {
507        for s in row.cells.keys() {
508            if s.kind() == SymbolKind::External {
509                return *s;
510            }
511        }
512        if (tag.marker.kind() == SymbolKind::Slack || tag.marker.kind() == SymbolKind::Error)
513            && row.coefficient_for(tag.marker) < 0.0
514        {
515            return tag.marker;
516        }
517        if (tag.other.kind() == SymbolKind::Slack || tag.other.kind() == SymbolKind::Error)
518            && row.coefficient_for(tag.other) < 0.0
519        {
520            return tag.other;
521        }
522        Symbol::invalid()
523    }

525    /// Add the row to the tableau using an artificial variable.
526    ///
527    /// This will return false if the constraint cannot be satisfied.
528    fn add_with_artificial_variable(&mut self, row: &Row) -> Result<bool, InternalSolverError> {
529        // Create and add the artificial variable to the tableau
530        let art = Symbol::new(self.id_tick, SymbolKind::Slack);
531        self.id_tick += 1;
532        self.rows.insert(art, Box::new(row.clone()));
533        self.artificial = Some(row.clone());
534
535        // Optimize the artificial objective. This is successful
536        // only if the artificial objective is optimized to zero.
537        self.optimize(OptimizeTarget::Artificial)?;
538        let success = near_zero(self.artificial.as_ref().unwrap().constant);
539        self.artificial = None;
540
541        // If the artificial variable is basic, pivot the row so that
542        // it becomes basic. If the row is constant, exit early.
543        if let Some(mut row) = self.rows.remove(&art) {
544            if row.cells.is_empty() {
545                return Ok(success);
546            }
547            let entering = Solver::any_pivotable_symbol(&row); // never External
548            if entering.kind() == SymbolKind::Invalid {
549                return Ok(false); // unsatisfiable (will this ever happen?)
550            }
551            row.solve_for_symbols(art, entering);
552            self.substitute(entering, &row);
553            self.rows.insert(entering, row);
554        }
555
556        // Remove the artificial row from the tableau
557        for row in self.rows.values_mut() {
558            row.remove(art);
559        }
560        self.objective.remove(art);
561        Ok(success)
562    }

564    /// Substitute the parametric symbol with the given row.
565    ///
566    /// This method will substitute all instances of the parametric symbol
567    /// in the tableau and the objective function with the given row.
568    fn substitute(&mut self, symbol: Symbol, row: &Row) {
569        for (&other_symbol, other_row) in &mut self.rows {
570            let constant_changed = other_row.substitute(symbol, row);
571            if other_symbol.kind() == SymbolKind::External && constant_changed {
572                let v = self.var_for_symbol[&other_symbol];
573                // inline var_changed
574                if self.should_clear_changes {
575                    self.changed.clear();
576                    self.should_clear_changes = false;
577                }
578                self.changed.insert(v);
579            }
580            if other_symbol.kind() != SymbolKind::External && other_row.constant < 0.0 {
581                self.infeasible_rows.push(other_symbol);
582            }
583        }
584        self.objective.substitute(symbol, row);
585        if let Some(ref mut artificial) = self.artificial {
586            artificial.substitute(symbol, row);
587        }
588    }

590    /// Optimize the system for the given objective function.
591    ///
592    /// This method performs iterations of Phase 2 of the simplex method
593    /// until the objective function reaches a minimum.
594    fn optimize(&mut self, target: OptimizeTarget) -> Result<(), InternalSolverError> {
595        loop {
596            let entering = {
597                let objective = match target {
598                    OptimizeTarget::Objective => &self.objective,
599                    OptimizeTarget::Artificial => self.artificial.as_ref().unwrap(),
600                };
601                Solver::get_entering_symbol(objective)
602            };
603            if entering.kind() == SymbolKind::Invalid {
604                return Ok(());
605            }
606            let (leaving, mut row) = self
607                .get_leaving_row(entering)
608                .ok_or(InternalSolverError::ObjectiveUnbounded)?;
609            // pivot the entering symbol into the basis
610            row.solve_for_symbols(leaving, entering);
611            self.substitute(entering, &row);
612            if entering.kind() == SymbolKind::External && row.constant != 0.0 {
613                let v = self.var_for_symbol[&entering];
614                self.var_changed(v);
615            }
616            self.rows.insert(entering, row);
617        }
618    }

620    /// Optimize the system using the dual of the simplex method.
621    ///
622    /// The current state of the system should be such that the objective
623    /// function is optimal, but not feasible. This method will perform
624    /// an iteration of the dual simplex method to make the solution both
625    /// optimal and feasible.
626    fn dual_optimize(&mut self) -> Result<(), InternalSolverError> {
627        while let Some(leaving) = self.infeasible_rows.pop() {
628            let row = if let Entry::Occupied(entry) = self.rows.entry(leaving) {
629                if entry.get().constant < 0.0 {
630                    Some(entry.remove())
631                } else {
632                    None
633                }
634            } else {
635                None
636            };
637            if let Some(mut row) = row {
638                let entering = self.get_dual_entering_symbol(&row);
639                if entering.kind() == SymbolKind::Invalid {
640                    return Err(InternalSolverError::DualOptimizeFailed);
641                }
642                // pivot the entering symbol into the basis
643                row.solve_for_symbols(leaving, entering);
644                self.substitute(entering, &row);
645                if entering.kind() == SymbolKind::External && row.constant != 0.0 {
646                    let v = self.var_for_symbol[&entering];
647                    self.var_changed(v);
648                }
649                self.rows.insert(entering, row);
650            }
651        }
652        Ok(())
653    }

655    /// Compute the entering variable for a pivot operation.
656    ///
657    /// This method will return first symbol in the objective function which
658    /// is non-dummy and has a coefficient less than zero. If no symbol meets
659    /// the criteria, it means the objective function is at a minimum, and an
660    /// invalid symbol is returned.
661    /// Could return an External symbol
662    fn get_entering_symbol(objective: &Row) -> Symbol {
663        for (symbol, value) in &objective.cells {
664            if symbol.kind() != SymbolKind::Dummy && *value < 0.0 {
665                return *symbol;
666            }
667        }
668        Symbol::invalid()
669    }

671    /// Compute the entering symbol for the dual optimize operation.
672    ///
673    /// This method will return the symbol in the row which has a positive
674    /// coefficient and yields the minimum ratio for its respective symbol
675    /// in the objective function. The provided row *must* be infeasible.
676    /// If no symbol is found which meats the criteria, an invalid symbol
677    /// is returned.
678    /// Could return an External symbol
679    fn get_dual_entering_symbol(&self, row: &Row) -> Symbol {
680        let mut entering = Symbol::invalid();
681        let mut ratio = f64::INFINITY;
682        for (symbol, value) in &row.cells {
683            if *value > 0.0 && symbol.kind() != SymbolKind::Dummy {
684                let coeff = self.objective.coefficient_for(*symbol);
685                let r = coeff / *value;
686                if r < ratio {
687                    ratio = r;
688                    entering = *symbol;
689                }
690            }
691        }
692        entering
693    }

695    /// Get the first Slack or Error symbol in the row.
696    ///
697    /// If no such symbol is present, and Invalid symbol will be returned.
698    /// Never returns an External symbol
699    fn any_pivotable_symbol(row: &Row) -> Symbol {
700        for symbol in row.cells.keys() {
701            if symbol.kind() == SymbolKind::Slack || symbol.kind() == SymbolKind::Error {
702                return *symbol;
703            }
704        }
705        Symbol::invalid()
706    }

708    /// Compute the row which holds the exit symbol for a pivot.
709    ///
710    /// This method will return an iterator to the row in the row map
711    /// which holds the exit symbol. If no appropriate exit symbol is
712    /// found, the end() iterator will be returned. This indicates that
713    /// the objective function is unbounded.
714    /// Never returns a row for an External symbol
715    fn get_leaving_row(&mut self, entering: Symbol) -> Option<(Symbol, Box<Row>)> {
716        let mut ratio = f64::INFINITY;
717        let mut found = None;
718        for (symbol, row) in &self.rows {
719            if symbol.kind() != SymbolKind::External {
720                let temp = row.coefficient_for(entering);
721                if temp < 0.0 {
722                    let temp_ratio = -row.constant / temp;
723                    if temp_ratio < ratio {
724                        ratio = temp_ratio;
725                        found = Some(*symbol);
726                    }
727                }
728            }
729        }
730        found.map(|s| (s, self.rows.remove(&s).unwrap()))
731    }

733    /// Compute the leaving row for a marker variable.
734    ///
735    /// This method will return an iterator to the row in the row map
736    /// which holds the given marker variable. The row will be chosen
737    /// according to the following precedence:
738    ///
739    /// 1) The row with a restricted basic varible and a negative coefficient for the marker with
740    ///    the smallest ratio of -constant / coefficient.
741    ///
742    /// 2) The row with a restricted basic variable and the smallest ratio of constant /
743    ///    coefficient.
744    ///
745    /// 3) The last unrestricted row which contains the marker.
746    ///
747    /// If the marker does not exist in any row, the row map end() iterator
748    /// will be returned. This indicates an internal solver error since
749    /// the marker *should* exist somewhere in the tableau.
750    fn get_marker_leaving_row(&mut self, marker: Symbol) -> Option<(Symbol, Box<Row>)> {
751        let mut r1 = f64::INFINITY;
752        let mut r2 = r1;
753        let mut first = None;
754        let mut second = None;
755        let mut third = None;
756        for (symbol, row) in &self.rows {
757            let c = row.coefficient_for(marker);
758            if c == 0.0 {
759                continue;
760            }
761            if symbol.kind() == SymbolKind::External {
762                third = Some(*symbol);
763            } else if c < 0.0 {
764                let r = -row.constant / c;
765                if r < r1 {
766                    r1 = r;
767                    first = Some(*symbol);
768                }
769            } else {
770                let r = row.constant / c;
771                if r < r2 {
772                    r2 = r;
773                    second = Some(*symbol);
774                }
775            }
776        }
777        first.or(second).or(third).and_then(|s| {
778            if s.kind() == SymbolKind::External && self.rows[&s].constant != 0.0 {
779                let v = self.var_for_symbol[&s];
780                self.var_changed(v);
781            }
782            self.rows.remove(&s).map(|r| (s, r))
783        })
784    }

786    /// Remove the effects of a constraint on the objective function.
787    fn remove_constraint_effects(&mut self, constraint: &Constraint, tag: &Tag) {
788        if tag.marker.kind() == SymbolKind::Error {
789            self.remove_marker_effects(tag.marker, constraint.strength().value());
790        }
791        if tag.other.kind() == SymbolKind::Error {
792            self.remove_marker_effects(tag.other, constraint.strength().value());
793        }
794    }

796    /// Remove the effects of an error marker on the objective function.
797    fn remove_marker_effects(&mut self, marker: Symbol, strength: f64) {
798        if let Some(row) = self.rows.get(&marker) {
799            self.objective.insert_row(row, -strength);
800        } else {
801            self.objective.insert_symbol(marker, -strength);
802        }
803    }

805    /// Test whether a row is composed of all dummy variables.
806    fn all_dummies(row: &Row) -> bool {
807        for symbol in row.cells.keys() {
808            if symbol.kind() != SymbolKind::Dummy {
809                return false;
810            }
811        }
812        true

815    /// Get the stored value for a variable.
816    ///
817    /// Normally values should be retrieved and updated using `fetch_changes`, but this method can
818    /// be used for debugging or testing.
819    pub fn get_value(&self, v: Variable) -> f64 {
820        self.var_data
821            .get(&v)
822            .and_then(|s| self.rows.get(&s.1).map(|r| r.constant))
823            .unwrap_or(0.0)
824    }
