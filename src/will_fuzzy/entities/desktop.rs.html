8/// base parsing of the file. probably should add more safety checks lol
9pub fn parse_desktop_inis() -> Vec<DesktopEntity>{
10    
11    // uh. maybe. idk just wanna be sure
12    if !cfg!(target_os = "linux") {
13        panic!("This program must be run on Linux!");
14    }
15    
16
17
18    let home_dir = std::env::var("HOME").unwrap_or_else(|_| "".into());
19    let desktop_paths_as_per_arch_wiki = vec![
20        format!("{}/.local/share/applications", home_dir), // User apps
21        "/usr/share/applications".to_string(),             // System apps
22        "/usr/local/share/applications".to_string(),       // Local system apps
23    ];
24
25    let entities: Vec<DesktopEntity> = desktop_paths_as_per_arch_wiki.iter()
26    .map(Path::new)
27    // 1. Turn the list of dirs into a stream of entries
28    .flat_map(|path| fs::read_dir(path).ok().into_iter().flatten())
29    .flatten() // Flatten the DirEntry results
30    // 2. Filter for .desktop files
31    .filter(|entry| entry.path().extension().and_then(|s| s.to_str()) == Some("desktop"))
32    // 3. Try to parse each file
33    .filter_map(|entry| DesktopEntity::from_file(&entry.path()))
34    // 4. Collect into your final Vec
35    .collect();
36    entities
37}

42pub struct DesktopEntity{
43    /// id = filename.desktop
44    id: String,
45    /// name = "Executable Desktop Nice Name" - "Strawberry"
46    name: String,
47    /// generic_name = ""  - "Strawberry Music Player"
48    generic_name: String,
49    /// desc = "description of it"
50    desc: String,
51    /// exec = "exec commands + args" - "strawberry %U"
52    exec: String,
53    /// tags<Vec Str> = []  - "\[AudioVideo\]\[Player\]\[Qt\]\[Audio\];" 
54    tags: Vec<String>,
55    /// freq? todo
56    launch_count: i64,
57    precompute_str: String,
58}

60impl FuzzyCandidate for DesktopEntity {
61    fn search_targets(&self) -> Vec<ScoreTarget> {
62        let mut targets = Vec::with_capacity(1+self.tags.len());
63        //generic w/ penalty
64        targets.push(
65            ScoreTarget::new(&self.name, 1.0, false)
66        );
67        for tag in &self.tags {
68            targets.push(
69            ScoreTarget::new(tag, 0.1, true)
70            );
71        }
72        targets
73    }
74    fn exec(&self) -> String{
75        self.exec.to_string()
76    }
77    fn usage_bonus(&self) -> i64 {
78        //tweak this heavy LOL
79        (self.launch_count as f64 * 1.2) as i64 
80    }
81    fn display_candidate(&self) -> String {
82        format!("{} ({})", self.name, self.exec)
83    }
84    fn display_text(&self) -> &str{
85        &self.precompute_str
86    }
87}

89impl DesktopEntity {
90    pub fn from_file(path: &Path) -> Option<Self> {
91        // Load the INI file
92        let conf = Ini::load_from_file(path).ok()?;
93        let section = conf.section(Some("Desktop Entry"))?;
94        let get_str = |key: &str| -> String {
95            section.get(key).unwrap_or("").to_string()
96        };
97        let name = get_str("Name");
98        let generic_name = get_str("GenericName");
99        let precompute_str = format!("{} - {}", name, generic_name);
100        let tags: Vec<String> = section
101            .get("Categories")
102            .map(|s| {
103                s.split(';')
104                    .filter(|tag| !tag.is_empty())
105                    .map(|tag| tag.to_string())
106                    .collect()
107            })
108        .unwrap_or_default();
109
110        Some(DesktopEntity {
111            id: path.file_name()?.to_string_lossy().to_string(),
112            name,
113            generic_name, 
114            desc: get_str("Comment"),
115            exec: get_str("Exec"),
116            tags,
117            launch_count: 0,
118            precompute_str,
119        })
120    }
121}
