61    fn process_key(&mut self, key: KeyEvent) {
62        match key.code {
63            KeyCode::Char(c)   => self.kp_basic_char(c),
64            KeyCode::Backspace => self.kp_backspace(),
65            // KeyCode::Enter     => self.toggles[0] = !self.toggles[0],
66            KeyCode::Enter     => self.kp_enter(),
67            KeyCode::Up        => self.kp_arrow_up(),
68            KeyCode::Down      => self.kp_arrow_down(),
69            KeyCode::Esc       => self.exit = true,
70            _ => {}
71        }

80    fn kp_enter(&mut self) {
81        if let Some(res) = self.session.current_results().get(self.hover_index.saturating_sub(1)) {
82            let exec_str = res.item.exec();
83            // non exec, toggle [2] which is just display text
84            if exec_str == "\0" {
85                self.toggles[2] = !self.toggles[2];
86            }
87            else{
88            if self.toggles[1] || exec_str == "\0" {
89                self.toggles[1] = false;
90            }
91            else{
92                self.toggles[1] = !self.toggles[1];
93            }
94            }
95
96            
97        }
98    }

102    pub (crate) fn sync_scroll(&mut self, num_line:usize){
103        // core logic:
104        // if no results, dont do anything early return 
105        if self.session.num_results == 0 {
106            self.scroll_index = 0;
107            self.list_state.select(None);
108            return;
109        }
110        // results? start by tracking current hovered index
111        // sub 1 b/c hover is one indexed for 0 marking none
112        let new_index = self.hover_index.saturating_sub(1);
113        // Ensure we don't select past the actual results
114        // set the furthest down to the length -1 (b/c len isnt 0 indexed) 
115        let max_possible_index = self.session.num_results.saturating_sub(1);
116        let clamped_index = new_index.min(max_possible_index);
117
118        // actually scroll
119        // upwards: cursor goes above top line, move up. straightforward
120        if clamped_index < self.scroll_index {
121            self.scroll_index = clamped_index;
122
123        //downward: if we hit the number visible + scroll index = bottom line of the code
124        } else if clamped_index >= self.scroll_index + num_line {
125            // we do the clampted + 1 to go down, and - num_line b/c 
126            // of the weird range scrlling thing we do
127            self.scroll_index = clamped_index + 1 - num_line;
128        }
129        // b/c we only are "rendering" a slice for optimization sake, we need to pretend for calcs
130        // that the whole list is all there, needs to be manually handled as opposed to just the
131        // basic select 
132        self.list_state.select(Some(clamped_index-self.scroll_index));
133
134    }
