61    pub fn one_step_calc(&self, target: &[u8], query: &[u8]) -> i64{
62        // eprintln!("{}",String::from_utf8(target.to_vec()).unwrap());
63        // eprintln!("target: {:?}  -  query: {:?}",target,query);

65        // let s = String::from_utf8((&target).to_vec()).expect("Found invalid UTF-8");
66        // refactor that actually works
67        // process psudo:
68        // if either empty, fail 
69        // ideally, fail if we dont match any bounds/start
70        // maybe? fail off first char not match. but more the any bounds
71        // prev iter matched usize var
72        // var score
73        // while over both,  
74        // check if identical
75        //    if match{
76        //    if start of string (index 1?) 
77        //              start bonus!
78        //    else if, check if prev matched. 
79        //        add consec bonus
80        //    else if prev is delimin
81        //         add delim bonus
82        //    update last matched index
83        //    advance query
84        //   }
85        //  regardless of match, advance target:
86        //}
87        // if we reach end of query, return score. (want to ensure target didnt end early
88        // i think? ran odd without)
89        if target.is_empty() || query.is_empty() { return 0 }
90        let mut prev_match_index=0;
91        let mut amt_consec = 1;
92        let mut score:i64 = 0;
93        let mut target_index=0;
94        let mut query_index=0;
95        while target_index < target.len() && query_index < query.len(){
96            let t_byte = target[target_index];
97            let q_byte = query[query_index];
98            //match, orr if either is a seperator(match)
99            if t_byte == q_byte || (is_sep(t_byte) && is_sep(q_byte)) {
100                // b/c we advance both together, or target alone, any target_index ==0 means its on
101                // first. so, apply bonus start
102                if target_index == 0 {
103                    score+=self.bonus_start;
104                }// if its not, its at least past 1. so we can safely check for -1 without extra
105                 // conditonal:
106                else{
107                    // check if past is a delim, if so, give delim bonus
108                    // such a finicky if, might be faster this way?
109                    // DELIM BONUS!
110                    // unsafe { std::arch::asm!(";# LLVM-MCA-BEGIN"); }
111                    let mask = SEPARATOR_MAP[target[target_index - 1] as usize];
112                    score += mask & self.bonus_bound;
113                    // score += -(is_sep(target[target_index - 1]) as i64) & self.bonus_bound;
114                    // unsafe { std::arch::asm!(";# LLVM-MCA-END"); }
115                     
116                    // CONSEC BONUS
117                    if target_index == prev_match_index + 1 {
118                        amt_consec+=5;
119                        score += self.bonus_consec+amt_consec;
120                    }
121                    else{ amt_consec = 0}
122                }
123                prev_match_index = target_index;
124
125                // advance query if match expr
126                query_index +=1;   
127            }
128            target_index +=1;
129        }
130        let mask = ((query_index == query.len()) as i64).wrapping_neg();
131        mask & score
132    }

135static SEPARATOR_MAP: [i64; 256] = {
136    let mut map = [0i64; 256];
137    map[b' ' as usize] = -1; // -1 is all 1s in bits
138    map[b'-' as usize] = -1;
139    map[b'_' as usize] = -1;
140    map[b'/' as usize] = -1;
141    map[b'\\' as usize] = -1;
142    map[b'.' as usize] = -1;
143    map
144};
145#[inline(always)]
146fn is_sep(b: u8) -> bool {
